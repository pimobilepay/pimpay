// -------------------------
// Enums
// -------------------------
enum TransactionStatus {
  PENDING
  SUCCESS
  FAILED
  CANCELLED
  REFUNDED
}

enum TransactionType {
  TRANSFER
  DEPOSIT
  WITHDRAWAL
  PAYMENT
  FEE
  REFUND
}

enum PaymentMethod {
  WALLET     // internal wallet (π)
  USD_BALANCE // fiat balance
  CARD
  EXTERNAL   // external pi wallet or other
  QR
}

enum Currency {
  PI
  USD
  EUR
  // add more if needed
}

enum KycStatus {
  NONE
  PENDING
  APPROVED
  REJECTED
}

enum NotificationType {
  TRANSACTION
  SYSTEM
  PROMOTION
  SECURITY
}

enum SupportStatus {
  OPEN
  PENDING
  CLOSED
}

// -------------------------
// Models
// -------------------------

model User {
  id             String        @id @default(uuid())
  email          String        @unique
  phone          String?       @unique
  name           String?
  fullName       String?
  password       String?       // hashed password (nullable for social/accounts without password)
  createdAt      DateTime      @default(now())
  updatedAt      DateTime      @updatedAt

  // Security
  pinHash        String?       // hashed PIN (bcrypt)
  biometricEnabled Boolean     @default(false)
  twoFactorEnabled Boolean     @default(false) // e.g. OTP by SMS/email

  // Relations
  profile        Profile?
  wallets        Wallet[]      // one-to-many (user can have multiple wallet entries - e.g. different currencies)
  transactions   Transaction[] // all transactions where user is sender or receiver (see fields in Transaction)
  qrPayments     QRPayment[]   // QR codes generated by user
  notifications  Notification[]
  messages       Message[]     // support messages or chats
  devices        Device[]      // registered devices for biometric/push
  kyc            Kyc?
  settings       Setting?      // user preferences

  @@map("users")
}

model Profile {
  id          String   @id @default(uuid())
  userId      String   @unique
  user        User     @relation(fields: [userId], references: [id], onDelete: Cascade)
  avatarUrl   String?
  bio         String?
  dob         DateTime?
  country     String?
  language    String?   @default("fr")
  createdAt   DateTime  @default(now())
  updatedAt   DateTime  @updatedAt

  @@map("profiles")
}

model Wallet {
  id          String   @id @default(uuid())
  userId      String
  user        User     @relation(fields: [userId], references: [id], onDelete: Cascade)
  currency    Currency @default(PI)
  balance     Decimal  @default("0") @db.Decimal(30, 8)
  blockedBalance Decimal @default("0") @db.Decimal(30, 8) // reserved funds
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt

  transactions Transaction[]  // relation from Transaction.walletId

  @@index([userId, currency])
  @@map("wallets")
}

// Generic transaction record used for transfers, deposits, withdrawals, payments, fees...
model Transaction {
  id             String             @id @default(uuid())
  reference      String             @unique // external reference / txid
  fromUserId     String?            // sender (nullable for deposits)
  toUserId       String?            // receiver (nullable for withdrawals)
  fromWalletId   String?            // optional wallet id of sender
  toWalletId     String?            // optional wallet id of receiver
  amount         Decimal            @db.Decimal(30, 8)
  fee            Decimal?           @db.Decimal(30, 8)
  currency       Currency           @default(PI)
  type           TransactionType
  method         PaymentMethod
  status         TransactionStatus  @default(PENDING)
  description    String?
  metadata       Json?              // arbitrary metadata (payment provider data, raw response, etc)
  createdAt      DateTime           @default(now())
  updatedAt      DateTime           @updatedAt

  // Relations
  fromUser       User?              @relation("tx_from_user", fields: [fromUserId], references: [id], onDelete: SetNull)
  toUser         User?              @relation("tx_to_user", fields: [toUserId], references: [id], onDelete: SetNull)

  fromWallet     Wallet?            @relation("tx_from_wallet", fields: [fromWalletId], references: [id], onDelete: SetNull)
  toWallet       Wallet?            @relation("tx_to_wallet", fields: [toWalletId], references: [id], onDelete: SetNull)

  qrPayment      QRPayment?         @relation(fields: [reference], references: [reference], onDelete: SetNull)

  @@index([fromUserId])
  @@index([toUserId])
  @@index([status])
  @@map("transactions")
}

// QR payments (generated QR codes to receive payments)
model QRPayment {
  id           String   @id @default(uuid())
  reference    String   @unique // used as reference for linking to Transaction (can be same as tx reference)
  userId       String
  user         User     @relation(fields: [userId], references: [id], onDelete: Cascade)
  amount       Decimal? @db.Decimal(30, 8) // optional suggested amount
  currency     Currency @default(PI)
  label        String?  // description or purpose
  expiresAt    DateTime?
  active       Boolean  @default(true)
  createdAt    DateTime @default(now())

  transactions Transaction[] @relation("qr_transactions") // transactions that used this QR

  @@index([userId])
  @@map("qr_payments")
}

// Notifications system
model Notification {
  id          String           @id @default(uuid())
  userId      String
  user        User             @relation(fields: [userId], references: [id], onDelete: Cascade)
  type        NotificationType @default(TRANSACTION)
  title       String
  message     String
  data        Json?            // optional json payload
  read        Boolean          @default(false)
  createdAt   DateTime         @default(now())

  @@index([userId, read])
  @@map("notifications")
}

// Messages / Support tickets
model Message {
  id          String     @id @default(uuid())
  userId      String?    // the user who sent the message (nullable for system)
  user        User?      @relation(fields: [userId], references: [id], onDelete: SetNull)
  subject     String?
  body        String
  attachments Json?      // array of urls / metadata
  isSystem    Boolean    @default(false)
  createdAt   DateTime   @default(now())

  @@index([userId])
  @@map("messages")
}

// Registered devices for push / biometric devices
model Device {
  id            String   @id @default(uuid())
  userId        String
  user          User     @relation(fields: [userId], references: [id], onDelete: Cascade)
  deviceId      String   // client device identifier
  platform      String?  // android / ios / web
  pushToken     String?  // for push notifications
  lastSeenAt    DateTime @default(now())
  createdAt     DateTime @default(now())
  updatedAt     DateTime @updatedAt

  @@index([userId, deviceId])
  @@map("devices")
}

// KYC information
model Kyc {
  id          String   @id @default(uuid())
  userId      String   @unique
  user        User     @relation(fields: [userId], references: [id], onDelete: Cascade)
  status      KycStatus @default(NONE)
  documentType String?
  documentNumber String?
  documentFiles Json?   // array of file URLs / metadata
  reviewedBy  String?   // admin id
  reviewedAt  DateTime?
  createdAt    DateTime @default(now())
  updatedAt    DateTime @updatedAt

  @@map("kyc")
}

// User-specific settings / preferences
model Setting {
  id           String  @id @default(uuid())
  userId       String  @unique
  user         User    @relation(fields: [userId], references: [id], onDelete: Cascade)
  language     String  @default("fr")
  locale       String? 
  pushEnabled  Boolean @default(true)
  createdAt    DateTime @default(now())
  updatedAt    DateTime @updatedAt

  @@map("settings")
}

// Support tickets (optional)
model SupportTicket {
  id          String    @id @default(uuid())
  userId      String
  user        User      @relation(fields: [userId], references: [id], onDelete: Cascade)
  subject     String
  status      SupportStatus @default(OPEN)
  messages    Message[] // messages linked to ticket
  createdAt   DateTime  @default(now())
  updatedAt   DateTime  @updatedAt

  @@map("support_tickets")
}

// Optional: audit logs for important events
model AuditLog {
  id          String   @id @default(uuid())
  userId      String?
  user        User?    @relation(fields: [userId], references: [id], onDelete: SetNull)
  action      String
  meta        Json?
  ip          String?
  createdAt   DateTime @default(now())

  @@map("audit_logs")
}
